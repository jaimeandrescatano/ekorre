 10 /* Funcion generadora de la tabla */
 11 function makeGrid (evt) {
 12     /* Defino las variables */
 13     var marranito;
 14     var perrito;
 15     var lineaTabla = "";
 16     var i = 1;
 17 
 18     /* Evito que se desaparezca la tabla generada */
 19     evt.preventDefault();
 20 
 21     /* Elimino la tabla existente si es que el usuario ya la ha definido antes */
 22     $("#pixel_canvas").find("tr").remove();
 23 
 24     /* Obtengo los valores del form */
 25     marranito = $("#input_height").val();
 26     perrito = $("#input_width").val();
 27 
 28     // console.log(marranito);
 29     // console.log(perrito);
 30 
 31     /* Genero la tabla */
 32     /* -- Genero los <td> usando ciclo "for" */
 33     for (var x = 1; x <= perrito; x++) {
 34         lineaTabla = lineaTabla+"<td>Hola!</td>";
 35     }
 36 
 37     /* -- Genero los <tr> usando ciclo "while" */
 38     while (i <= marranito) {
 39         $("#pixel_canvas").append("<tr>"+lineaTabla+"</tr>");
 40         i = i+1;
 41     }
 42 }

 44 /* Funcion que toma el color seleccionado */
 45 function elegirColor (evt) {
 46     /* Obtengo el color */
 47     micoTiti = $("#colorPicker").val();
 48 
 49     /* Asigno el color a la celda en la que se hizo click */
 50     $('#pixel_canvas').on('click', 'td', function() {
 51         $(this).css('background-color', micoTiti );
 52     });
 53     console.log(micoTiti);
 54 }

 56 /* Creo mi funcion */
 57 function miPerrito () {
 58     /* Obtengo los valores de la cantidad de filas y de columnas de la tabla */
 59     $("#sizePicker").on("submit", makeGrid);
 60     $("#colorPicker").on("change", elegirColor);
 61 }
 62 
 63 /* Paso mi funcion al objeto de jQuery */
 64 $(miPerrito);
